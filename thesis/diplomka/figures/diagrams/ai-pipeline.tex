% Sekvenční diagram pipeline umělé inteligence pro rozpoznání jídla z fotografie.
% Aktéři: UI -- AiPipelineService -- AiService -- Externí API.
% Self-cally na AiPipelineService: sanitizace vstupu, sestavení požadavku, confidence gate.
% Self-call na AiService: parsování odpovědi.
% Confidence gate je modelováno UML alt fragmentem se dvěma větvemi podle hodnoty spolehlivosti.
% Detaily implementace jednotlivých kroků jsou popsány v textu sekce 4.5.
\begin{tikzpicture}[
	font=\small\sffamily,
	x=1cm, y=1cm,
	actor/.style={
		draw,
		rounded corners=1.5pt,
		minimum height=12mm,
		minimum width=24mm,
		align=center,
		fill=black!8,
	},
	lifeline/.style={dashed, black!50},
	activation/.style={draw=black, fill=black!15, line width=0.5pt},
	msg/.style={-{Latex[length=2mm]}, thick},
	ret/.style={-{Latex[length=2mm]}, thick, densely dashed},
	mlbl/.style={font=\small\sffamily, midway, above, sloped=false, fill=white, inner sep=1pt, outer sep=1.5pt},
	mlblguard/.style={font=\small\sffamily, midway, above, sloped=false, fill=white, inner sep=2pt, outer sep=1.5pt, align=center},
	selflbl/.style={anchor=west, font=\small\sffamily, fill=white, inner sep=1pt},
	frame/.style={draw=black, line width=0.4pt},
	frlabel/.style={draw=black, fill=white, anchor=north west, inner sep=2pt, font=\scriptsize\sffamily\bfseries, line width=0.4pt},
]

% --- Aktéři ---
\node[actor] (a1) at (0,0)    {User\\Interface\strut};
\node[actor] (a2) at (3.5,0)  {\texttt{AiPipeline}\strut};
\node[actor] (a4) at (7.0,0)  {\texttt{AiService}\strut};
\node[actor] (a5) at (10.5,0) {\texttt{OpenAI~API}\strut};

% --- Lifelines ---
\foreach \a in {a1,a2,a4,a5}
	\draw[lifeline] (\a.south) -- ($(\a.south)+(0,-9.0)$);

% --- Aktivační pruhy ---
% Uživatelské rozhraní je aktivní (čeká) po celou dobu pipeline.
% UI bar je o něco delší než Pipeline bar -- UI je aktivní těsně před odesláním volání
% a těsně po přijetí výsledku, takže obklopuje Pipeline bar zhora i zdola.
\draw[activation] ($(a1.south)+(-0.15,-0.35)$) rectangle ($(a1.south)+(0.15,-8.55)$);
% AiPipelineService je aktivní po dobu skutečného zpracování -- bar začíná těsně pod UI barem
% a končí těsně nad jeho spodním okrajem, čímž je nesený UI barem.
\draw[activation] ($(a2.south)+(-0.15,-0.5)$) rectangle ($(a2.south)+(0.15,-8.35)$);
% AiService je aktivní od volání generate(...) do návratu AiResponse.
\draw[activation] ($(a4.south)+(-0.15,-3.35)$) rectangle ($(a4.south)+(0.15,-6.7)$);
% Externí API je aktivní mezi přijetím POST a odesláním JSON odpovědi.
\draw[activation] ($(a5.south)+(-0.15,-3.95)$) rectangle ($(a5.south)+(0.15,-5.05)$);

% --- Sekvence zpráv ---
% 1. UI -> Pipeline: vstup
\draw[msg] ($(a1.south)+(0.15,-0.5)$) -- ($(a2.south)+(-0.15,-0.5)$)
	node[mlbl, xshift=0.0cm] {\texttt{analyzeMeal()}};

% 2. Pipeline self-loop: sanitizace vstupu
\draw[msg] ($(a2.south)+(0.15,-1.2)$) .. controls +(0.7,0) and +(0.7,0) .. ($(a2.south)+(0.15,-1.7)$);
\node[selflbl] at ($(a2.south)+(0.8,-1.45)$) {input sanitization};

% 3. Pipeline self-loop: sestavení požadavku (zahrnuje base64 kódování fotografie i sestavení promptu)
\draw[msg] ($(a2.south)+(0.15,-2.3)$) .. controls +(0.7,0) and +(0.7,0) .. ($(a2.south)+(0.15,-2.8)$);
\node[selflbl] at ($(a2.south)+(0.8,-2.55)$) {request assembly};

% 4. Pipeline -> AiService: generate
\draw[msg] ($(a2.south)+(0.15,-3.35)$) -- ($(a4.south)+(-0.15,-3.35)$)
	node[mlbl, xshift=0.4cm] {\texttt{generateResponse()}};

% 5. AiService -> Externí API: POST
\draw[msg] ($(a4.south)+(0.15,-3.95)$) -- ($(a5.south)+(-0.15,-3.95)$)
	node[mlbl, xshift=1.2cm] {\texttt{POST /v1/chat/completions}};

% 6. Externí API -> AiService: JSON odpověď
\draw[ret] ($(a5.south)+(-0.15,-5.05)$) -- ($(a4.south)+(0.15,-5.05)$)
	node[mlbl, xshift=0.0cm] {JSON response};

% 7. AiService self-loop: parsování odpovědi
\draw[msg] ($(a4.south)+(0.15,-5.65)$) .. controls +(0.7,0) and +(0.7,0) .. ($(a4.south)+(0.15,-6.15)$);
\node[selflbl] at ($(a4.south)+(0.8,-5.9)$) {parser};

% 8. AiService -> Pipeline: AiResponse
\draw[ret] ($(a4.south)+(-0.15,-6.7)$) -- ($(a2.south)+(0.15,-6.7)$)
	node[mlbl, xshift=0.0cm] {\texttt{AiResponse}};

% 9. Pipeline self-loop: confidence gate
\draw[msg] ($(a2.south)+(0.15,-7.25)$) .. controls +(0.7,0) and +(0.7,0) .. ($(a2.south)+(0.15,-7.75)$);
\node[selflbl] at ($(a2.south)+(0.8,-7.5)$) {confidence gate};

% 10. Pipeline -> UI: výsledek (forma závisí na hodnotě confidence, popsáno v textu sekce 4.5.3)
\draw[ret] ($(a2.south)+(-0.15,-8.35)$) -- ($(a1.south)+(0.15,-8.35)$)
	node[mlbl, xshift=0.1cm] {\texttt{AiAnalysisResult}};

\end{tikzpicture}
