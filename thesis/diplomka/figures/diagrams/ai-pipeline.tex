		align=center,
		fill=black!8,
	},
	lifeline/.style={dashed, black!50},
	msg/.style={-{Latex[length=2mm]}, thick},

% --- Aktéři ---
\node[actor] (a1) at (0,0)   {Uživatelské\\rozhraní};
\node[actor] (a2) at (3.0,0) {\texttt{AiPipeline}\\\texttt{Service}};

% 8. AiService -> Pipeline
