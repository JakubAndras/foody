% Vrstvený mobilní klient (5 vrstev) + lokální SQLite + 3 externí API.
\begin{tikzpicture}[
	font=\small\sffamily,
	node distance=4mm,
	layer/.style={
		draw,
		rounded corners=1.5pt,
		minimum height=8mm,
		minimum width=58mm,
		align=center,
		fill=black!4,
	},
	clientbox/.style={
		draw,
		thick,
		rounded corners=3pt,
		inner sep=4mm,
		fill=black!2,
	},
	ext/.style={
		draw,
		rounded corners=1.5pt,
		minimum height=11mm,
		minimum width=32mm,
		align=center,
		fill=black!6,
	},
	extInactive/.style={
		draw,
		dashed,
		rounded corners=1.5pt,
		minimum height=11mm,
		minimum width=32mm,
		align=center,
		fill=black!2,
	},
	arrInactive/.style={-{Latex[length=2mm]}, thick, dashed},
	db/.style={
		draw,
		cylinder,
		shape border rotate=90,
		aspect=0.25,
		minimum height=14mm,
		minimum width=28mm,
		align=center,
		fill=black!6,
	},
	arr/.style={-{Latex[length=2mm]}, thick},
	dataflow/.style={font=\scriptsize\sffamily, midway, sloped, above, inner sep=1pt},
]

% Vrstvy mobilního klienta
\node[layer] (view) {Prezentační vrstva (View)\\\scriptsize \texttt{lib/screens/}, \texttt{lib/widgets/}};
\node[layer, below=of view] (ctrl) {Kontrolery (ViewModel)\\\scriptsize \texttt{lib/controller/} -- GetX};
\node[layer, below=of ctrl] (svc) {Služby a repozitáře\\\scriptsize \texttt{lib/services/}, \texttt{Repository}};
\node[layer, below=of svc] (model) {Modely\\\scriptsize \texttt{lib/model/} -- \texttt{json\_serializable}};
\node[layer, below=of model] (data) {Datová vrstva\\\scriptsize \texttt{lib/database/} (Floor) -- \texttt{lib/network/} (Dio)};

% Obal mobilního klienta -- kreslíme na background layer, aby nezakryl vnitřní vrstvy
\begin{scope}[on background layer]
\node[clientbox, fit=(view)(ctrl)(svc)(data)(model), label={[font=\small\sffamily\bfseries, anchor=south]north:Mobilní klient (Flutter)}] (client) {};
\end{scope}

% Lokální databáze pod klientem
\node[db, below=10mm of client] (sqlite) {Lokální\\databáze\\\scriptsize SQLite};

% Externí služby (vpravo) -- těsnější odsazení po odstranění popisků šipek
\node[ext, right=22mm of ctrl, yshift=1mm] (openai) {OpenAI\\\scriptsize Chat Completions API};
\node[extInactive, below=of openai] (gemini) {Google Gemini\\\scriptsize\itshape (alternativní)};
\node[ext, below=of gemini] (off) {Open Food Facts\\\scriptsize Product API};

% Šipky: mobilní klient ↔ SQLite (start zespodu klienta)
\draw[arr, <->] (client.south -| sqlite.north) -- (sqlite.north);

% Šipky: vrstva služeb / sítě ↔ externí API (sběrnice ven z klienta)
\coordinate (busTop) at ($(client.east)+(6mm,0)$);
\coordinate (busOpenai) at (busTop |- openai.west);
\coordinate (busGemini) at (busTop |- gemini.west);
\coordinate (busOff)    at (busTop |- off.west);

\draw[arr, <-] (client.east |- gemini.west) -- (busTop |- gemini.west);
\draw[arr] (busTop |- gemini.west) -- (busOpenai) -- (openai.west);
\draw[arrInactive] (busTop |- gemini.west) -- (busGemini) -- (gemini.west);
\draw[arr] (busTop |- gemini.west) -- (busOff)    -- (off.west);

\end{tikzpicture}
